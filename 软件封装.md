```bash
#!/bin/bash
# 获取当前脚本所在绝对路径，绝对不会路径出错
SCRIPT_DIR=$(cd "$(dirname "${BASH_SOURCE[0]}")" &>/dev/null && pwd)

# 加载本包内自带的ROS环境（不用系统安装ROS！）
source ${SCRIPT_DIR}/devel/setup.bash

# 优先使用本包内自带所有库，不调用系统库

# 启动你的Qt主程序

```

给脚本执行权限：
```bash

```
