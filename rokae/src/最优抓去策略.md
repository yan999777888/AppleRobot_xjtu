struct GraspCandidate {

/**
 * [通用 Look-At 算法]
 * 给定一个进给方向(Z轴)，计算出使夹爪开口(X轴)尽量水平的欧拉角
 */
vector<double> calcPoseFromVector(Vector3d approach_dir) {
    // 1. Z轴: 进给方向 (归一化)
    Vector3d z_axis = approach_dir.normalized();

    // 2. 期望的辅助轴: 我们希望夹爪的 X 轴(开口)尽量水平，且垂直于 Z
    // 世界坐标系下的 "下" 是 (0, 0, -1)
    // 或者世界坐标系下的 "侧面" 是 (0, 1, 0)
    // 这里我们假设：期望 X 轴尽量指向 Y 方向 (侧抓)
    Vector3d preferred_dir(0, 1, 0); 
    
    // 如果 Z 轴和 preferred_dir 平行，叉乘会失效，换一个辅助轴
    if (abs(z_axis.dot(preferred_dir)) > 0.95) {
        preferred_dir = Vector3d(0, 0, -1);
    }

    // 3. 计算 X 轴: Z cross preferred -> X (生成的X垂直于Z和preferred)
    Vector3d x_axis = z_axis.cross(preferred_dir).normalized();

    // 4. 计算 Y 轴: Z cross X -> Y (右手定则)
    Vector3d y_axis = z_axis.cross(x_axis).normalized();

    // *注意*: 上面算出的 y_axis 是真正的 Y。
    // 如果你的机械臂末端定义不同(比如 Z 是进给, Y 是开口)，请调整 xyz 的赋值顺序
    
    // 5. 构建旋转矩阵
    Matrix3d R;
    R.col(0) = x_axis;
    R.col(1) = y_axis;
    R.col(2) = z_axis;

    // 6. 转欧拉角 (Rokae SDK 只要这一步对上就行，通常是 ZYX)
    Vector3d euler = R.eulerAngles(2, 1, 0); 
    return {euler[2], euler[1], euler[0]}; // Rx, Ry, Rz
}

/**
 * [斐波那契球面采样 + 评分系统]
 * 生成并筛选最优策略
 */
vector<GraspCandidate> planGraspStrategies(Vector3d center_pos) {
    vector<GraspCandidate> candidates;
    
    // 黄金分割角
    double phi = M_PI * (3.0 - sqrt(5.0)); 

    for (int i = 0; i < NUM_SAMPLES; ++i) {
        // --- A. 采样 (Sampling) ---
        // 生成球面上的点 (y是从1到-1，我们只取上半球，所以映射一下范围)
        // 这里的坐标系是球体局部坐标系
        double yi = 1 - (i / float(NUM_SAMPLES - 1)) * 2; // -1 to 1
        double radius = sqrt(1 - yi * yi);
        double theta = phi * i;

        double xi = cos(theta) * radius;
        double zi = sin(theta) * radius;

        // 构造进给向量: 假设在机器人基座坐标系中，Z是高，X是前
        // 我们希望从球面上一点，指向球心
        Vector3d vec_sphere(xi, zi, yi); // 映射: 球y -> 机器人z(高), 球x -> 机器人x, 球z -> 机器人y
        Vector3d approach = -vec_sphere.normalized();

        // --- B. 约束过滤 (Constraint Filtering) ---
        // 规则1: 绝对不能从下往上抓 (Z > 0.1)，防止撞到底座或树枝
        if (approach.z() > 0.1) continue;
        
        // 规则2: 不能从机器人后面往前抓 (X > 0)，必须是从前往后或侧面
        if (approach.x() > 0.1) continue;

        // --- C. 评分 (Scoring / Cost Function) ---
        double score = 100.0;

        // 因子1: 水平优先 (Z分量越小越好)
        // 权重: 30
        score -= abs(approach.z()) * 30.0;

        // 因子2: 正对优先 (X分量越接近-1越好)
        // 权重: 40
        // (approach.x + 1) 在 0~1 之间，越小说明越接近 -1
        score -= (1.0 + approach.x()) * 40.0;

        // --- D. 生成候选 ---
        GraspCandidate task;
        task.id = i;
        task.approach_vec = approach;
        task.rpy = calcPoseFromVector(approach);
        task.score = score;
        
        candidates.push_back(task);
    }

    // 按分数从高到低排序
    sort(candidates.begin(), candidates.end(), 
         [](const GraspCandidate& a, const GraspCandidate& b) { return a.score > b.score; });

    return candidates;
}

void waitForFinish(BaseRobot &robot, const std::string &traj_id, int index){
  std::error_code ec;
  while(true) {
    auto info = robot.queryEventInfo(Event::moveExecution, ec);
    if(ec) continue;
    auto _id = std::any_cast<std::string>(info.at(EventInfoKey::MoveExecution::ID));
    auto _idx = std::any_cast<int>(info.at(EventInfoKey::MoveExecution::WaypointIndex));
    if(auto _ec = std::any_cast<std::error_code>(info.at(EventInfoKey::MoveExecution::Error))) {
        // 运动报错直接返回
        return; 
    }
    if(_id == traj_id && _idx == index) return;
    std::this_thread::sleep_for(std::chrono::milliseconds(50));
  }
}

    else pub_serial.publish(gripper_open_cmd);
    ros::Duration(0.8).sleep();
}

Eigen::Matrix4d loadMatrix(const std::string& filePath){
    Eigen::Matrix4d matrix;
    std::ifstream file(filePath);
    if (!file.is_open()) throw std::runtime_error("File load failed: " + filePath);
    for (int i = 0; i < 4; ++i) for (int j = 0; j < 4; ++j) file >> matrix(i, j);
    file.close(); return matrix;
}

bool pickCallBack(task_assign::Target::Request& req, task_assign::Target::Response& resp) {
    if (!robot_ptr) return false;
    std::error_code ec;

    // 1. 坐标转换
    Vector4d p_cam(req.x, req.y, req.z * depth_scale, 1.0);
    Vector4d p_base = T_cam_to_base * p_cam;
    Vector3d target_pos = p_base.head<3>();

    ROS_INFO(">>> 收到目标: [%.3f, %.3f, %.3f]", target_pos.x(), target_pos.y(), target_pos.z());

    // 2. 规划策略 (Planner)
    auto strategies = planGraspStrategies(target_pos);
    ROS_INFO("生成了 %ld 个可行策略，正在尝试最优解...", strategies.size());

    if (strategies.empty()) {
        ROS_ERROR("无解：所有采样角度均不可行");
        resp.success = false;
        return true;
    }

    // 3. 尝试执行 Top-N 策略 (Execution Loop)
    bool success = false;
    string id;
    int max_attempts = 5; // 最多试前5名，避免浪费时间

    for (int i = 0; i < min((int)strategies.size(), max_attempts); ++i) {
        auto& task = strategies[i];
        
        ROS_INFO("尝试策略 #%d (Score: %.1f): Dir=[%.2f, %.2f, %.2f]", 
                 i+1, task.score, task.approach_vec.x(), task.approach_vec.y(), task.approach_vec.z());

        // 计算路径点
        Vector3d pre_pos = target_pos - task.approach_vec * DIST_PRE;
        
        // 安全高度检查
        if (pre_pos.z() < 0.15) {
            ROS_WARN(" -> 预备点过低，跳过");
            continue;
        }

        try {
            // (1) 移动到预备点 (使用 MoveL 或 MoveJ)
            controlGripper(false);
            MoveLCommand cmd_pre({pre_pos.x(), pre_pos.y(), pre_pos.z(), 
                                task.rpy[0], task.rpy[1], task.rpy[2]});
            cmd_pre.speed = 60; // 快
            
            // 发送指令
            robot_ptr->moveAppend({cmd_pre}, id, ec);
            robot_ptr->moveStart(ec);
            waitForFinish(*robot_ptr, id, 0);
            
            // 检查是否报错 (IK失败等)
            if (ec) throw std::runtime_error("IK Failed at Pre-Grasp");

            // (2) 进给 (Approach)
            MoveLCommand cmd_grasp({target_pos.x(), target_pos.y(), target_pos.z(), 
                                  task.rpy[0], task.rpy[1], task.rpy[2]});
            cmd_grasp.speed = 15; // 慢
            robot_ptr->moveAppend({cmd_grasp}, id, ec);
            robot_ptr->moveStart(ec);
            waitForFinish(*robot_ptr, id, 0);

            if (ec) throw std::runtime_error("Collision or IK Failed at Grasp");

            // (3) 抓取
            controlGripper(true);

            // (4) 撤退 (Retreat) - 沿原路返回一段距离
            Vector3d ret_pos = target_pos - task.approach_vec * DIST_RET;
            MoveLCommand cmd_ret({ret_pos.x(), ret_pos.y(), ret_pos.z(), 
                                task.rpy[0], task.rpy[1], task.rpy[2]});
            cmd_ret.speed = 40;
            robot_ptr->moveAppend({cmd_ret}, id, ec);
            robot_ptr->moveStart(ec);
            waitForFinish(*robot_ptr, id, 0);

            // 成功！
            success = true;
            break; 

        } catch (...) {
            ROS_WARN(" -> 策略执行失败，尝试下一个...");
            // 如果是在进给过程中报错，可能需要复位，这里简化处理
            continue; 
        }
    }

    if (success) {
        // (5) 放置 (Drop)
        MoveJCommand cmd_drop({0.65, 0.2, 0.0, -M_PI/2, M_PI/2, -M_PI/2});
        robot_ptr->moveAppend({cmd_drop}, id, ec);
        robot_ptr->moveStart(ec);
        waitForFinish(*robot_ptr, id, 0);
        
        controlGripper(false);
        resp.success = true;
        ROS_INFO("任务圆满完成！");
    } else {
        resp.success = false;
        resp.message = "Optimization Failed";
        ROS_ERROR("所有策略均尝试失败");
    }
    return true;
}

代码的核心逻辑解析（写在论文里的话）
    1. 动态采样 (Dynamic Sampling):
代码没有硬编码“从左抓”或“从前抓”。planGraspStrategies 函数在苹果的球面上生成了 50 个点。无论苹果在空间中什么位置，这些点都是相对于苹果中心的，保证了覆盖率。
    2. 约束过滤 (Constraint Awareness):
if (approach.z() > 0.1) continue;
这行代码体现了环境认知。我们知道从下往上抓通常会碰到下方的树枝或底座，所以直接在数学层面剔除这些解。
    3. 最优性 (Optimality):
score 的计算结合了两个因子：
        ◦ 姿态舒适度： 机械臂在伸直去抓的时候，末端姿态越水平，关节通常越舒服（远离奇异点）。
        ◦ 接近度： 优先选择正对着基座的方向，因为这通常意味着最短路径。
    4. 鲁棒执行 (Robust Execution):
try-catch 块和 max_attempts 循环。机器人会先计算分数最高的策略（比如 Top 1），如果 Top 1 因为 IK 无解（比如苹果太远或者关节限位）报错了，它不会停止，而是自动尝试 Top 2。这就是智能的体现。

// =================== 核心数学算法更新 ===================

/**
 * [通用 Look-At 算法]
 * 给定一个进给方向(Z轴)，计算出使夹爪开口(X轴)尽量水平的欧拉角
 */
vector<double> calcPoseFromVector(Vector3d approach_dir) {
    // 1. Z轴: 进给方向 (指向苹果中心)
    Vector3d z_axis = approach_dir.normalized();

    // 2. 期望的辅助轴: 
    // 我们希望夹爪的 X 轴(开口)尽量水平，且垂直于 Z。
    // 
    // 特殊情况处理：
    // 如果是从下往上抓 (Z轴向上)，我们通常希望夹爪不要“扭着”，
    // 而是保持自然的仰角。这里依然使用 (0,1,0) 作为参考侧面是比较通用的。
    Vector3d preferred_dir(0, 1, 0); 
    
    // 如果 Z 轴和 preferred_dir 平行 (极其罕见的侧面垂直进给)，换一个辅助轴
    if (abs(z_axis.dot(preferred_dir)) > 0.95) {
        preferred_dir = Vector3d(1, 0, 0); // 换成 X 轴做参考
    }

    // 3. 计算 X 轴: Z cross preferred -> X
    // 这样生成的 X 轴大致是水平的
    Vector3d x_axis = z_axis.cross(preferred_dir).normalized();

    // 4. 计算 Y 轴: Z cross X -> Y (右手定则)
    Vector3d y_axis = z_axis.cross(x_axis).normalized();

    // 5. 构建旋转矩阵
    Matrix3d R;
    R.col(0) = x_axis;
    R.col(1) = y_axis;
    R.col(2) = z_axis;

    // 6. 转欧拉角 (Rokae ZYX)
    Vector3d euler = R.eulerAngles(2, 1, 0); 
    return {euler[2], euler[1], euler[0]}; 
}

/**
 * [更新版：支持从下往上抓取]
 */
vector<GraspCandidate> planGraspStrategies(Vector3d center_pos) {
    vector<GraspCandidate> candidates;
    
    // 黄金分割角
    double phi = M_PI * (3.0 - sqrt(5.0)); 
    
    // 采样点数
    int num_samples = 60; //稍微增加采样密度

    for (int i = 0; i < num_samples; ++i) {
        // --- A. 采样 (Sampling) ---
        // 生成球面上的点 (-1 到 1，覆盖整个球)
        double yi = 1 - (i / float(num_samples - 1)) * 2; 
        double radius = sqrt(1 - yi * yi);
        double theta = phi * i;

        double xi = cos(theta) * radius;
        double zi = sin(theta) * radius;

        // 构造进给向量: 
        // 映射: 球y -> 机器人z(高), 球x -> 机器人x, 球z -> 机器人y
        Vector3d vec_sphere(xi, zi, yi); 
        
        // 关键：进给向量是从球面指向球心
        Vector3d approach = -vec_sphere.normalized();

        // --- B. 约束过滤 (Constraint Filtering) ---
        
        // 规则1: 禁止从上往下抓 (Top-Down)
        // 意味着进给向量的 Z 分量不能是负数 (或者不能小于很小的值)
        // 实际操作中，稍微有一点点向下(-0.1)可能还可以接受，但为了安全，我们设为 -0.05
        if (approach.z() < -0.05) continue; 
        
        // 规则2: 禁止从后往前抓 (X > 0)
        // 机械臂底座在后面，不能把手伸到苹果后面往回掏，那样会撞到苹果树的主干
        if (approach.x() > 0.1) continue;

        // --- C. 评分 (Scoring / Cost Function) ---
        double score = 100.0;

        // 因子1: 奖励仰角 (Bottom-Up) 和 水平 (Horizontal)
        // approach.z() > 0 表示从下往上，这是安全的
        // 我们给 Z > 0 的姿态加分，或者至少不减分
        if (approach.z() > 0) {
            score += approach.z() * 20.0; // 越是向上托举，分数越高 (避开上方树枝)
        } else {
            // 轻微向下的 (如果规则1放宽了)，扣分
            score -= abs(approach.z()) * 50.0; 
        }

        // 因子2: 正对优先 (X分量越接近-1越好)
        // 保持不变，还是希望正对着去抓，路径最短
        score -= (1.0 + approach.x()) * 30.0;

        // 因子3: 高度惩罚 (可选)
        // 如果苹果太低，从下往上抓可能会让机械臂关节撞到底座
        // center_pos 是苹果的绝对高度
        if (center_pos.z() < 0.4 && approach.z() > 0.5) {
            // 如果苹果很低(小于0.4m)，且还要大仰角抓，扣分，因为手腕可能着地
            score -= 50.0;
        }

        // --- D. 生成候选 ---
        GraspCandidate task;
        task.id = i;
        task.approach_vec = approach;
        task.rpy = calcPoseFromVector(approach);
        task.score = score;
        
        candidates.push_back(task);
    }

    // 排序
    sort(candidates.begin(), candidates.end(), 
         [](const GraspCandidate& a, const GraspCandidate& b) { return a.score > b.score; });

    return candidates;
}
代码修改解读 (用于理解或写论文)
    1. 安全空间的定义变更：
        ◦ 原逻辑： 上方是空的，下方可能撞底座 -> 优先水平/俯视。
        ◦ 现逻辑（果园实战）： 上方有树枝，下方是空的 -> 优先水平/仰视（Bottom-Up）。
        ◦ 代码中通过 if (approach.z() < -0.05) continue; 这一行强力屏蔽了所有可能压断树枝的动作。
    2. 仰角抓取的特殊性：
        ◦ 当 approach.z() > 0 时，机械臂末端是斜向上的。
        ◦ 我们在评分函数中增加了 score += approach.z() * 20.0。这意味着，如果一个策略是 “从下方45度角切入”，它的得分可能会超过 “水平切入”。
        ◦ 这非常符合采摘逻辑：顺着苹果下垂的方向逆流而上，最容易让果柄落入夹爪缝隙中。
    3. 防撞底座保护：
        ◦ 从下往上抓有一个隐患：如果苹果长得太低（比如离地只有 30cm），机械臂要想“仰视”去抓，手肘或手腕可能会戳到地面或底座。
        ◦ 所以我加了一个动态评分调整：if (center_pos.z() < 0.4 && approach.z() > 0.5) score -= 50.0;。如果是低处的苹果，系统会自动降级仰角策略，强迫机械臂选择水平抓取。
